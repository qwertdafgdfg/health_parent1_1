	2)  套餐预约占比饼形图
		sql 

			-- 只查询有的
				
				SELECT s.name,COUNT(s.id) 'value' FROM t_order  o , t_setmeal s  WHERE s.`id`= o.`setmeal_id`
				GROUP BY  s.`name`

				-- 查询全部

				SELECT s.name,COUNT(o.id) FROM  t_setmeal s  LEFT JOIN   t_order  o ON  s.`id`= o.`setmeal_id`  
				GROUP BY  s.`name`


				SELECT s.name,COUNT(o.id) FROM   t_order  o   RIGHT JOIN  t_setmeal s ON  s.`id`= o.`setmeal_id`  
				GROUP BY  s.`name`
